function interpretLua(code) {
    const lines = code.split('\n');
    let variables = {};
    let localVars = {};
    let tables = {};
    let functions = {};
    let i = 0;

    function evaluate(expr) {
        expr = expr.trim();
        if ((expr.startsWith('"') && expr.endsWith('"')) || (expr.startsWith("'") && expr.endsWith("'"))) {
            return expr.slice(1, -1);
        }
        if (expr.startsWith('[[') && expr.endsWith(']]')) {
            return expr.slice(2, -2);
        }
        if (!isNaN(parseFloat(expr)) && isFinite(expr)) {
            return parseFloat(expr);
        }
        if (expr === 'true') return true;
        if (expr === 'false') return false;
        if (expr === 'nil') return null;
        if (localVars.hasOwnProperty(expr)) return localVars[expr];
        if (variables.hasOwnProperty(expr)) return variables[expr];
        if (tables.hasOwnProperty(expr)) return tables[expr];

        const dotMatch = expr.match(/^(.+)\.(.+)$/);
        if (dotMatch) {
            const tableName = dotMatch[1].trim();
            const key = dotMatch[2].trim();
            const table = tables[tableName] || variables[tableName] || localVars[tableName];
            if (table && typeof table === 'object') return table[key];
        }

        const bracketMatch = expr.match(/^(.+)\[(.+)\]$/);
        if (bracketMatch) {
            const tableName = bracketMatch[1].trim();
            const key = evaluate(bracketMatch[2].trim());
            const table = tables[tableName] || variables[tableName] || localVars[tableName];
            if (table && typeof table === 'object') return table[key];
        }

        if (expr.includes('..')) {
            const parts = expr.split('..').map(p => evaluate(p.trim()));
            return parts.join('');
        }

        try {
            let evalExpr = expr;
            for (let v in localVars) {
                evalExpr = evalExpr.replace(new RegExp('\\b' + v + '\\b', 'g'), JSON.stringify(localVars[v]));
            }
            for (let v in variables) {
                evalExpr = evalExpr.replace(new RegExp('\\b' + v + '\\b', 'g'), JSON.stringify(variables[v]));
            }
            evalExpr = evalExpr.replace(/~= /g, '!== ');
            evalExpr = evalExpr.replace(/\/\/.*$/g, '');
            if (/^[0-9+\-*/^%().\s]+$/.test(evalExpr)) {
                evalExpr = evalExpr.replace(/\^/g, '**');
                return Function('"use strict"; return (' + evalExpr + ')')();
            }
            return Function('"use strict"; return (' + evalExpr + ')')();
        } catch(e) {
            return expr;
        }
    }

    while (i < lines.length) {
        let line = lines[i].trim();
        if (!line || line.startsWith('--')) { i++; continue; }
        const commentIdx = line.indexOf(' --');
        if (commentIdx > 0) line = line.substring(0, commentIdx).trim();
        if (line.startsWith('--[[')) {
            while (i < lines.length && !lines[i].includes(']]')) i++;
            i++; continue;
        }

        if (line.startsWith('local ')) {
            const rest = line.substring(6).trim();
            if (rest.startsWith('function ')) {
                const funcMatch = rest.match(/function\s+(\w+)\s*\(([^)]*)\)/);
                if (funcMatch) {
                    const funcName = funcMatch[1];
                    const params = funcMatch[2].split(',').map(p => p.trim()).filter(p => p);
                    let funcBody = [];
                    i++;
                    let depth = 1;
                    while (i < lines.length && depth > 0) {
                        const bodyLine = lines[i].trim();
                        if (bodyLine.startsWith('function ') || bodyLine.startsWith('if ') || bodyLine.startsWith('for ') || bodyLine.startsWith('while ') || bodyLine.startsWith('repeat')) depth++;
                        if (bodyLine === 'end') depth--;
                        if (depth > 0) funcBody.push(lines[i]);
                        i++;
                    }
                    localVars[funcName] = { type: 'function', params, body: funcBody };
                    continue;
                }
            }
            const tableMatch = rest.match(/^(\w+)\s*=\s*\{([^}]*)\}/);
            if (tableMatch) {
                const name = tableMatch[1];
                const content = tableMatch[2];
                const table = {};
                if (content.trim()) {
                    const items = content.split(',').map(s => s.trim()).filter(s => s);
                    items.forEach((item, idx) => {
                        if (item.includes('=')) {
                            const [k, v] = item.split('=').map(s => s.trim());
                            table[k] = evaluate(v);
                        } else {
                            table[idx + 1] = evaluate(item);
                        }
                    });
                }
                localVars[name] = table;
                tables[name] = table;
                i++;
                continue;
            }
            if (rest.includes('=')) {
                const eqIdx = rest.indexOf('=');
                const names = rest.substring(0, eqIdx).split(',').map(n => n.trim());
                const values = rest.substring(eqIdx + 1).split(',').map(v => evaluate(v.trim()));
                names.forEach((name, idx) => {
                    localVars[name] = values[idx] !== undefined ? values[idx] : null;
                });
            } else {
                rest.split(',').forEach(n => { localVars[n.trim()] = null; });
            }
            i++;
            continue;
        }

        if (line.startsWith('function ')) {
            const funcMatch = line.match(/function\s+(\w+)\s*\(([^)]*)\)/);
            if (funcMatch) {
                const funcName = funcMatch[1];
                const params = funcMatch[2].split(',').map(p => p.trim()).filter(p => p);
                let funcBody = [];
                i++;
                let depth = 1;
                while (i < lines.length && depth > 0) {
                    const bodyLine = lines[i].trim();
                    if (bodyLine.startsWith('function ') || bodyLine.startsWith('if ') || bodyLine.startsWith('for ') || bodyLine.startsWith('while ') || bodyLine.startsWith('repeat')) depth++;
                    if (bodyLine === 'end') depth--;
                    if (depth > 0) funcBody.push(lines[i]);
                    i++;
                }
                functions[funcName] = { params, body: funcBody };
                continue;
            }
        }

        if (line.startsWith('if ')) {
            const condition = line.substring(3).replace(/then$/, '').trim();
            const condResult = evaluate(condition);
            let ifBody = [];
            let elseBody = [];
            let inElse = false;
            i++;
            let depth = 1;
            while (i < lines.length && depth > 0) {
                const bodyLine = lines[i].trim();
                if (bodyLine.startsWith('if ')) depth++;
                if (bodyLine === 'end') {
                    depth--;
                    if (depth === 0) { i++; break; }
                }
                if (depth === 1 && bodyLine.startsWith('elseif ')) { inElse = true; i++; continue; }
                if (depth === 1 && bodyLine === 'else') { inElse = true; i++; continue; }
                if (depth > 0) {
                    if (inElse) elseBody.push(lines[i]);
                    else ifBody.push(lines[i]);
                }
                i++;
            }
            if (condResult) {
                ifBody.forEach(bline => {
                    const trimmed = bline.trim();
                    if (trimmed.startsWith('print(')) {
                        const arg = trimmed.substring(6, trimmed.length - 1);
                        addOutput('output', evaluate(arg));
                    }
                });
            } else {
                elseBody.forEach(bline => {
                    const trimmed = bline.trim();
                    if (trimmed.startsWith('print(')) {
                        const arg = trimmed.substring(6, trimmed.length - 1);
                        addOutput('output', evaluate(arg));
                    }
                });
            }
            continue;
        }

        if (line.startsWith('for ')) {
            const forMatch = line.match(/for\s+(\w+)\s*=\s*(.+)\s*,\s*(.+)\s+do/);
            if (forMatch) {
                const varName = forMatch[1];
                const start = evaluate(forMatch[2]);
                const end = evaluate(forMatch[3]);
                let loopBody = [];
                i++;
                while (i < lines.length && lines[i].trim() !== 'end') {
                    loopBody.push(lines[i]);
                    i++;
                }
                i++;
                for (let j = start; j <= end; j++) {
                    localVars[varName] = j;
                    loopBody.forEach(bline => {
                        const trimmed = bline.trim();
                        if (trimmed.startsWith('print(')) {
                            const arg = trimmed.substring(6, trimmed.length - 1);
                            let result = evaluate(arg);
                            if (typeof result === 'string') {
                                result = result.replace(new RegExp('\\.\\.' + varName, 'g'), j.toString());
                                result = result.replace(new RegExp(varName + '\\.\\.', 'g'), j.toString());
                            }
                            addOutput('output', result);
                        }
                    });
                }
                continue;
            }
            const pairsMatch = line.match(/for\s+(\w+),?\s*(\w*)?\s+in\s+pairs\((.+)\)\s+do/);
            if (pairsMatch) {
                const keyVar = pairsMatch[1];
                const valVar = pairsMatch[2] || '';
                const tableName = pairsMatch[3];
                const table = tables[tableName] || variables[tableName] || localVars[tableName] || {};
                let loopBody = [];
                i++;
                while (i < lines.length && lines[i].trim() !== 'end') {
                    loopBody.push(lines[i]);
                    i++;
                }
                i++;
                for (let k in table) {
                    if (valVar) localVars[valVar] = table[k];
                    localVars[keyVar] = k;
                    loopBody.forEach(bline => {
                        const trimmed = bline.trim();
                        if (trimmed.startsWith('print(')) {
                            const arg = trimmed.substring(6, trimmed.length - 1);
                            addOutput('output', evaluate(arg));
                        }
                    });
                }
                continue;
            }
            const ipairsMatch = line.match(/for\s+(\w+),\s*(\w+)\s+in\s+ipairs\((.+)\)\s+do/);
            if (ipairsMatch) {
                const idxVar = ipairsMatch[1];
                const valVar = ipairsMatch[2];
                const tableName = ipairsMatch[3];
                const table = tables[tableName] || variables[tableName] || localVars[tableName] || {};
                let loopBody = [];
                i++;
                while (i < lines.length && lines[i].trim() !== 'end') {
                    loopBody.push(lines[i]);
                    i++;
                }
                i++;
                let idx = 1;
                while (table[idx] !== undefined) {
                    localVars[idxVar] = idx;
                    localVars[valVar] = table[idx];
                    loopBody.forEach(bline => {
                        const trimmed = bline.trim();
                        if (trimmed.startsWith('print(')) {
                            const arg = trimmed.substring(6, trimmed.length - 1);
                            addOutput('output', evaluate(arg));
                        }
                    });
                    idx++;
                }
                continue;
            }
        }

        if (line.startsWith('while ')) {
            const condition = line.substring(6).replace(/do$/, '').trim();
            let loopBody = [];
            i++;
            while (i < lines.length && lines[i].trim() !== 'end') {
                loopBody.push(lines[i]);
                i++;
            }
            i++;
            let safety = 0;
            while (evaluate(condition) && safety < 100) {
                safety++;
                loopBody.forEach(bline => {
                    const trimmed = bline.trim();
                    if (trimmed.startsWith('print(')) {
                        const arg = trimmed.substring(6, trimmed.length - 1);
                        addOutput('output', evaluate(arg));
                    }
                    if (trimmed.includes('=') && !trimmed.startsWith('local')) {
                        const eqIdx = trimmed.indexOf('=');
                        const name = trimmed.substring(0, eqIdx).trim();
                        const val = evaluate(trimmed.substring(eqIdx + 1).trim());
                        if (localVars.hasOwnProperty(name)) localVars[name] = val;
                        else variables[name] = val;
                    }
                });
            }
            continue;
        }

        if (line.startsWith('repeat')) {
            let loopBody = [];
            i++;
            while (i < lines.length && !lines[i].trim().startsWith('until ')) {
                loopBody.push(lines[i]);
                i++;
            }
            const untilLine = lines[i].trim();
            const condition = untilLine.substring(6).trim();
            i++;
            let safety = 0;
            do {
                safety++;
                loopBody.forEach(bline => {
                    const trimmed = bline.trim();
                    if (trimmed.startsWith('print(')) {
                        const arg = trimmed.substring(6, trimmed.length - 1);
                        addOutput('output', evaluate(arg));
                    }
                });
            } while (!evaluate(condition) && safety < 100);
            continue;
        }

        if (line.includes('=') && !line.startsWith('if') && !line.startsWith('for') && !line.startsWith('while')) {
            const eqIdx = line.indexOf('=');
            const name = line.substring(0, eqIdx).trim();
            const value = line.substring(eqIdx + 1).trim();
            if (name.includes('.')) {
                const parts = name.split('.');
                const tableName = parts[0];
                const key = parts[1];
                const table = tables[tableName] || variables[tableName] || localVars[tableName];
                if (table && typeof table === 'object') {
                    table[key] = evaluate(value);
                }
            } else if (name.includes('[')) {
                const bracketMatch = name.match(/^(.+)\[(.+)\]$/);
                if (bracketMatch) {
                    const tableName = bracketMatch[1].trim();
                    const key = evaluate(bracketMatch[2].trim());
                    const table = tables[tableName] || variables[tableName] || localVars[tableName];
                    if (table && typeof table === 'object') {
                        table[key] = evaluate(value);
                    }
                }
            } else {
                if (localVars.hasOwnProperty(name)) {
                    localVars[name] = evaluate(value);
                } else {
                    variables[name] = evaluate(value);
                }
            }
            i++;
            continue;
        }

        if (line.startsWith('print(')) {
            const arg = line.substring(6, line.length - 1);
            const result = evaluate(arg);
            addOutput('output', result);
            i++;
            continue;
        }

        const funcCallMatch = line.match(/^(\w+)\s*\(([^)]*)\)$/);
        if (funcCallMatch && functions[funcCallMatch[1]]) {
            const funcName = funcCallMatch[1];
            const args = funcCallMatch[2].split(',').map(a => evaluate(a.trim()));
            const func = functions[funcName];
            const savedVars = {...localVars};
            func.params.forEach((p, idx) => {
                localVars[p] = args[idx];
            });
            func.body.forEach(bline => {
                const trimmed = bline.trim();
                if (trimmed.startsWith('return ')) {
                    const ret = evaluate(trimmed.substring(7));
                    addOutput('output', ret);
                } else if (trimmed.startsWith('print(')) {
                    const arg = trimmed.substring(6, trimmed.length - 1);
                    addOutput('output', evaluate(arg));
                }
            });
            localVars = savedVars;
            i++;
            continue;
        }

        if (line.startsWith('type(')) {
            const arg = line.substring(5, line.length - 1);
            const val = evaluate(arg);
            let typeName = typeof val;
            if (val === null) typeName = 'nil';
            if (Array.isArray(val) || (typeof val === 'object' && val !== null)) typeName = 'table';
            if (typeof val === 'boolean') typeName = 'boolean';
            addOutput('output', typeName);
            i++;
            continue;
        }

        if (line.startsWith('tostring(')) {
            const arg = line.substring(9, line.length - 1);
            addOutput('output', String(evaluate(arg)));
            i++;
            continue;
        }

        if (line.includes('string.')) {
            const strMatch = line.match(/string\.(\w+)\(([^)]+)\)/);
            if (strMatch) {
                const func = strMatch[1];
                const args = strMatch[2].split(',').map(a => evaluate(a.trim()));
                let result;
                if (func === 'upper') result = String(args[0]).toUpperCase();
                else if (func === 'lower') result = String(args[0]).toLowerCase();
                else if (func === 'gsub') result = String(args[0]).replace(new RegExp(args[1], 'g'), args[2]);
                else if (func === 'sub') result = String(args[0]).substring(args[1] - 1, args[2]);
                else if (func === 'len') result = String(args[0]).length;
                else if (func === 'find') result = String(args[0]).indexOf(args[1]) + 1;
                else if (func === 'format') result = args[0].replace(/%s/g, () => args[1]).replace(/%d/g, () => args[1]);
                else result = line;
                if (line.startsWith('print(')) {
                    addOutput('output', result);
                }
            }
            i++;
            continue;
        }

        if (line.includes('math.')) {
            const mathMatch = line.match(/math\.(\w+)\(([^)]*)\)/);
            if (mathMatch) {
                const func = mathMatch[1];
                const args = mathMatch[2] ? mathMatch[2].split(',').map(a => evaluate(a.trim())) : [];
                let result;
                if (func === 'sqrt') result = Math.sqrt(args[0]);
                else if (func === 'random') result = args.length > 1 ? Math.floor(Math.random() * (args[1] - args[0] + 1)) + args[0] : Math.random();
                else if (func === 'floor') result = Math.floor(args[0]);
                else if (func === 'ceil') result = Math.ceil(args[0]);
                else if (func === 'abs') result = Math.abs(args[0]);
                else if (func === 'max') result = Math.max(...args);
                else if (func === 'min') result = Math.min(...args);
                else if (func === 'pi') result = Math.PI;
                else if (func === 'sin') result = Math.sin(args[0]);
                else if (func === 'cos') result = Math.cos(args[0]);
                else if (func === 'tan') result = Math.tan(args[0]);
                else if (func === 'rad') result = args[0] * Math.PI / 180;
                else if (func === 'deg') result = args[0] * 180 / Math.PI;
                else result = 'math.' + func;
                if (line.startsWith('print(')) {
                    addOutput('output', result);
                }
            } else if (line.includes('math.pi')) {
                if (line.startsWith('print(')) {
                    addOutput('output', Math.PI);
                }
            }
            i++;
            continue;
        }

        if (line.includes('table.')) {
            const tableMatch = line.match(/table\.(\w+)\(([^)]+)\)/);
            if (tableMatch) {
                const func = tableMatch[1];
                const args = tableMatch[2].split(',').map(a => evaluate(a.trim()));
                let result;
                if (func === 'concat') {
                    const arr = args[0];
                    const sep = args[1] || ', ';
                    if (Array.isArray(arr)) result = arr.join(sep);
                    else result = Object.values(arr).join(sep);
                } else if (func === 'insert') {
                    const arr = args[0];
                    const val = args[1];
                    if (Array.isArray(arr)) arr.push(val);
                    result = arr;
                } else if (func === 'remove') {
                    const arr = args[0];
                    const idx = args[1];
                    if (Array.isArray(arr)) result = arr.splice(idx - 1, 1)[0];
                } else if (func === 'sort') {
                    const arr = args[0];
                    if (Array.isArray(arr)) arr.sort((a, b) => a - b);
                    result = arr;
                } else {
                    result = line;
                }
                if (line.startsWith('print(')) {
                    addOutput('output', result);
                }
            }
            i++;
            continue;
        }

        if (line.includes('setmetatable')) {
            const metaMatch = line.match(/setmetatable\(([^,]+),\s*(\w+)\)/);
            if (metaMatch) {
                const tableName = metaMatch[1].trim();
                const metaName = metaMatch[2].trim();
                if (tables[metaName]) {
                    if (tables[tableName]) {
                        tables[tableName].__metatable = tables[metaName];
                    } else if (localVars[tableName]) {
                        localVars[tableName].__metatable = tables[metaName];
                    }
                }
            }
            i++;
            continue;
        }

        if (line.includes('coroutine.')) {
            const coMatch = line.match(/coroutine\.(\w+)\(([^)]*)\)/);
            if (coMatch) {
                const func = coMatch[1];
                if (func === 'create') {
                    addOutput('info', '🧵 Coroutine created (simulated)');
                } else if (func === 'resume') {
                    addOutput('info', '🧵 Coroutine resumed (simulated)');
                } else if (func === 'yield') {
                    addOutput('info', '🧵 Coroutine yielded (simulated)');
                }
            }
            i++;
            continue;
        }

        i++;
    }
}
