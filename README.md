	local tweenInfo = TweenInfo.new(

	)

		tween:Play()
		tween.Completed:Connect(function()
			main2Frame.Visible = false  -- إخفاء الإطار بعد انتهاء التويين
		end)

	button.MouseButton1Click:Connect(function()
		if main2Frame then  -- التحقق من وجود الإطار Main 2
			if main2Frame.Visible then
				hideMain2()  -- إخفاء الإطار مع أنيميشن

		Enum.EasingDirection.Out -- اتجاه التغيير (من البداية إلى النهاية)
	)

				-- إذا كان هذا هو الزر الذي تم الضغط عليه، طبق التويين للون الأحمر

			end
		end
	end

local function FLVQUNE_fake_script() -- ScriptGuiAR.RemoveLagButton.Remove Lag Script 
	local script = Instance.new('LocalScript', ScriptGuiAR.RemoveLagButton)

	local remove = script.Parent
	
	-- جدول لتخزين العناصر المحذوفة
	local removedEffects = {}
	local effectsActive = false -- حالة التحكم في التأثيرات
	
	remove.MouseButton1Click:Connect(function()
		-- الأنواع التي سيتم إزالتها لتحسين الأداء
		local effectTypes = {
			"PointLight", "SurfaceLight", "SpotLight", 
			"Smoke", "Fire", "ParticleEmitter", 
			"Beam", "Trail", "Explosion", 
			"Decal", "Texture", "MeshPart", "Terrain", "Script"
		}
	
		-- وظيفة لإزالة التأثيرات
		local function removeEffects()
			for _, obj in pairs(game:GetService("Workspace"):GetDescendants()) do
				for _, effectType in ipairs(effectTypes) do
					if obj:IsA(effectType) then
						table.insert(removedEffects, obj) -- تخزين العنصر المحذوف
						obj:Destroy() -- إزالة العنصر
					end
				end
			end
		end
	
		-- وظيفة لاستعادة التأثيرات المحذوفة
		local function restoreEffects()
			for _, obj in pairs(removedEffects) do
				if obj:IsA("Model") then
					obj.Parent = game:GetService("Workspace") -- استعادة العنصر إلى Workspace
				end
			end
			removedEffects = {} -- إعادة تعيين الجدول بعد الاستعادة
		end
	
		-- التبديل بين الإزالة والاستعادة
		if effectsActive then
			restoreEffects()
		else
			removeEffects()
		end
	
		effectsActive = not effectsActive -- تحديث الحالة
	end)
	
end
